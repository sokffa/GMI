function assignUser()
{
	global $connect;
    $entity = $_POST['entity'];
	$type = $_POST['typesel'];
	$code = $_POST['codesel'];
	$dop = $_POST['txtDp'];
	$dor = $_POST['txtDr'];
	$uid = $_POST['txtUid'];
	if(empty($dor))
		$dor='0000-00-00';
	if($entity == 1){
			//update avbl_qty
			$sql = "UPDATE hardwares 
					SET avbl_qty = avbl_qty - 1
					WHERE id = $type";
			$connect->query($sql);
			//update etat
			$sql = "UPDATE hard_unite 
					SET etat = 'Affecté'
					WHERE code = '".$code."'";
			$connect->query($sql);
			//
			$sql = "INSERT INTO affectation (entity, type, uid, doa, doe, bdate ,code)
					VALUES($entity, $type, $uid, '$dop', '$dor', NOW() ,'$code')";

			header('Location: ../assign');
							
		
	}else {
		//echo "SW";

	}
	
}
